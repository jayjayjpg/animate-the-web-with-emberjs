Drawing To The Canvas

// app/components/comic-panel.js
export default Ember.Component.extend({
  draw() {
    const frameNum = this.get('frameNum');
    const sourceX = this.get('frameHeight') * frameNum;
    const ctx = this.get('ctx');
    const canvasWidth = this.get('width');
    const canvasHeight = this.get('height');
    const img = this.get('imgObject');
    const destinationXY = 1000;
  }
})

Drawing To The Canvas

/* app/components/comic-panel.js */  
  draw() {
    /* .... */
    /* clear canvas from previous draw */
    ctx.clearRect(0, 0, canvasWidth, canvasHeight);

    /* draw image to the canvas */
    ctx.drawImage(img, 0, sourceX, destinationXY, 
                  canvasWidth, 0, 0, destinationXY, 
                  canvasHeight);
  }

      const maxFrameNumber = this.get('maxFrameNumber');
      if (this.get('frameNum') < maxFrameNumber) {
        this.incrementProperty('frameNum');
      }
      else {
         this.set('frameNum', 0);
      }
    }
  }

Controlling the animation


/* app/components/comic-panel/component.js */
export default Ember.Component.extend({
  timeline: document.timeline,
  timelineAnimations: Ember.computed('timeline', function(){
    return this.get('timeline').getAnimations();
  }),
  actions: {
    play() {
      this.get('timelineAnimations').play();
      this.set('isPlaying', true);
    },
    pause() {
      this.get('timelineAnimations').pause();
      this.set('isPlaying', false);
    }
